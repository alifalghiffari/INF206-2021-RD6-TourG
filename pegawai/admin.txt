CREATE TABLE `admin` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `nik` varchar(20) COLLATE utf8_unicode_ci NOT NULL,
 `nip` varchar(20) COLLATE utf8_unicode_ci NOT NULL,
 `gampong` varchar(30) COLLATE utf8_unicode_ci NOT NULL,
 `kode_pos` varchar(10) COLLATE utf8_unicode_ci NOT NULL,
 `password` varchar(255) COLLATE utf8_unicode_ci NOT NULL,
 `dibuat` datetime NOT NULL,
 `diubah` datetime NOT NULL,
 `status` enum('1','0') COLLATE utf8_unicode_ci NOT NULL DEFAULT '1',
 PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
